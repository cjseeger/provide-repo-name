Some simple JavaScript is below.



<canvas id="myCanvas" width="200" height="200"></canvas>

    <script>
        // 1. Get the canvas element from the HTML
        const canvas = document.getElementById("myCanvas");

        // 2. Get the 2D drawing context
        const ctx = canvas.getContext("2d");

        // 3. Set the fill color for the square
        ctx.fillStyle = "red";

        // 4. Draw a filled rectangle (which is a square if width and height are equal)
        // fillRect(x, y, width, height)
        // x and y define the top-left corner position
        // width and height define the size
        ctx.fillRect(50, 50, 100, 100);
    </script>
